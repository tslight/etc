** BEGIN PROCESS END
#+BEGIN_SRC snippet :tangle ~/.emacs.d/snippets/powershell-mode/begin_process_end
  # -*- mode: snippet -*-
  # name: begin_process_end
  # key: bpe
  # --
  begin {
      ${1:Write-Verbose} \"${2:Processing} ${3:}\"
  }

  process {
      $0
  }

  end {
      ${4:Write-Verbose} \"${5:Finished processing} ${6:}\"
  }
#+END_SRC
