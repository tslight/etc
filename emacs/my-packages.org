* BUFFERS

[[file:my-buffers.org][My Buffers Functions]]

#+BEGIN_SRC emacs-lisp
  (use-package my-buffers
    :after my-packages-minimal
    :commands
    my/fill-buffer
    my/indent-buffer
    my/toggle-buffer
    :bind
    ("C-`" . my/last-buffer)
    ("M-<escape>" . my/last-buffer)
    ("C-c M-i" . my/indent-buffer)
    ("C-c M-n" . my/nuke-buffers)
    ("C-c s" . my/save-buffers-silently)
    ("M-s s" . my/search-all-buffers)
    ("C-c t m" . (lambda () (interactive) (my/toggle-buffer "*Messages*")))
    ("C-c t s" . (lambda () (interactive) (my/toggle-buffer "*scratch*"))))
#+END_SRC
