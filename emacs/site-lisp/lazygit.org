#+begin_src emacs-lisp
  (defun my/filter-keys (list keys)
    "Filter a LIST of association lists by a list of KEYS."
    (mapcar (lambda (al)
              (mapcar (lambda (key)
                        (assoc key al))
                      keys))
            list))
#+end_src

** GET ASSOCIATION LIST

#+begin_src emacs-lisp
  (defun my/get-assoc-list (list key value)
    "Retrieve an association list from a LIST of association lists
  that has KEY VALUE pair."
    (car (cl-remove-if nil (mapcar (lambda (e)
                                  (if (equal value (cdr (assoc key e)))
                                      e))
                                list))))
#+end_src

* URL FUNCTIONS
#+begin_src emacs-lisp
  (defun my/url-link-to-next-page ()
    (goto-char (point-min))
    (let ((begin (re-search-forward "Link.*: <" nil t))
          (end (re-search-forward ">; rel=\"next\"" nil t)))
      (if (and begin end)
          (buffer-substring-no-properties begin (- end 13))
        nil)))

  (defun my/url-retrieve-paginated-bodies (url &optional headers items)
    "Retrieve resource body ITEMS from URL, with optional
  HEADERS. Supports key-based pagination - i.e) if returned headers
  have a link to the next page. Returns all pages bodies as a
  single string."
    (set-buffer
     (let ((url-request-method "GET")
           (url-request-extra-headers headers))
       (url-retrieve-synchronously url)))
    (let ((next-page (my/url-link-to-next-page)))
      (delete-region (point-min) url-http-end-of-headers)
      (let* ((retrieved-items (cons (buffer-string) items)))
        (if next-page
            (my/url-retrieve-paginated-bodies next-page headers retrieved-items)
          retrieved-items))))

  (defun my/flatten-json (items)
    "Take a list of json arrays as ITEMS and flatten them into a
  single json array."
    (concat "[" (mapconcat
                 (lambda (item)
                   (let*
                       ((trimmed (replace-regexp-in-string "^\\[\\]$" "" item))
                        (trimmed (replace-regexp-in-string "^\\[" "" trimmed))
                        (trimmed (if (equal item (car (last items)))
                                     (replace-regexp-in-string "\\]$" "" trimmed)
                                   (replace-regexp-in-string "\\]$" "," trimmed))))
                     trimmed))
                 items "")
            "]"))

  (defun my/url-parse-retrieved-json (url &optional headers)
    "Translate a list of json objects retrieved from URL with
  optional HEADERS into an association list."
    (let* ((json (my/url-retrieve-paginated-bodies url headers))
           (json-array-type 'list)
           (parsed-json (json-read-from-string (my/flatten-json json))))
      parsed-json))

  (defun my/url-view-retrieved-json (url &optional headers)
    "View the json retreived from URL, with optional HEADERS,
  properly formatted in a buffer."
    (let* ((items (my/url-retrieve-paginated-bodies url headers))
           (name (replace-regexp-in-string
                  "\\(^https://\\|.com/api/v4\\|?.*\\)" "" url))
           (name (replace-regexp-in-string "/" "-" name)))
      (switch-to-buffer (concat "*" name "*"))
      (erase-buffer)
      (insert (my/flatten-json items))
      (json-pretty-print-buffer)
      (goto-char (point-min))))

  (defun my/url-get-values (url keys &optional headers)
    "Retrieve values from KEYS from a list of URL json objects."
    (let ((response (my/url-parse-retrieved-json url headers)))
      (my/filter-keys response keys)))
#+end_src

* GIT FUNCTIONS
** CLONE OR PULL

#+begin_src emacs-lisp
  (defun my/clone-or-pull (repos path name url directory)
    "Prompt for a Github REPOS, then clone that repo to DIRECTORY.
  Using PATH, NAME & URL."
    (let* ((paths (mapcar (lambda (r) (cdr (assoc path r))) repos))
           (choice (completing-read "Repo to clone: " paths))
           (repo (my/get-assoc-list repos path choice))
           (cloneurl (cdr (assoc url repo)))
           (reponame (cdr (assoc name repo)))
           (dest (concat directory reponame)))
      (if (file-directory-p dest)
          (progn
            (message (string-trim (shell-command-to-string (concat "git -C " dest " pull"))))
            (magit-status-setup-buffer dest))
        (progn
          (message (concat "Cloning " cloneurl " to " dest "..."))
          (magit-clone-regular cloneurl dest nil)))))
#+end_src

#+begin_src emacs-lisp
  (defun my/gitlab-retriever (endpoint)
    "Retrieve resources from GitLab ENDPOINT."
    (interactive "sEnter an endpoint: ")
    (my/url-view-retrieved-json (concat "https://gitlab.com/api/v4/"
                                        endpoint
                                        "?pagination=keyset&per_page=100&"
                                        "order_by=id&sort=asc"
                                        "&owned=true&membership=true")
                                ;; https://stackoverflow.com/a/24188208
                                `(("PRIVATE-TOKEN" . ,my/gitlab-api-key))))

  (defun my/gitlab-get-values (endpoint keys)
    "Retrieve values from KEYS of GitLab ENDPOINT json resources."
    (my/url-get-values (concat "https://gitlab.com/api/v4/"
                               endpoint
                               "?pagination=keyset&per_page=100&"
                               "order_by=id&sort=asc"
                               "&owned=true&membership=true")
                       keys
                       ;; https://stackoverflow.com/a/24188208
                       `(("PRIVATE-TOKEN" . ,my/gitlab-api-key))))

  (defun my/gitlab-clone-or-pull (directory)
    "Clone or pull repo to DIRECTORY."
    (interactive "DDirectory to clone to: ")
    (my/clone-or-pull (my/gitlab-get-values "projects" (list 'path_with_namespace
                                                             'name
                                                             'ssh_url_to_repo))
                      'path_with_namespace
                      'name
                      'ssh_url_to_repo
                      directory))
