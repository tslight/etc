* BOOTSTRAP

Set up package.el, add melpa repository, add package autoloads, then install
[[https://github.com/jwiegley/use-package][use-package]], if it's not already.

Set some use-package variables, and then add our configuration file directory to
the [[help:load-path][load path]].

Finally load our package configuration, which calls use-package to set
everything else up.

#+BEGIN_SRC emacs-lisp
;; setup package.el
(require 'package)
(setq package-enable-at-startup nil)
(setq package--init-file-ensured t)
(setq package-archives
      '(("melpa" . "https://melpa.org/packages/")
        ("gnu" . "https://elpa.gnu.org/packages/")))
(package-initialize)

;; install use-package
(unless (package-installed-p 'use-package)
  (package-refresh-contents)
  (package-install 'use-package))

;; setup use-package
(require 'use-package)
(setq use-package-enable-imenu-support t
      use-package-verbose t)
#+END_SRC

* EXPAND REGION

https://spwhitton.name/blog/entry/expandregionlines/

#+BEGIN_SRC emacs-lisp
  (use-package expand-region
    :ensure t
    :bind
    ("C-=" . er/expand-region)
    ("C--" . er/contract-region)
    :config
    (defadvice er/expand-region (around fill-out-region activate)
      (if (or (not (region-active-p))
              (eq last-command 'er/expand-region))
          ad-do-it
        (if (< (point) (mark))
            (let ((beg (point)))
              (goto-char (mark))
              (end-of-line)
              (forward-char 1)
              (push-mark)
              (goto-char beg)
              (beginning-of-line))
          (let ((end (point)))
            (goto-char (mark))
            (beginning-of-line)
            (push-mark)
            (goto-char end)
            (end-of-line)
            (forward-char 1))))))
#+END_SRC

[[https://emacs.stackexchange.com/questions/2745/make-fill-paragraph-in-python-docstring-leave-the-triple-quotes-on-separate-line/2758][Docstring Fill Column Mode]]

#+BEGIN_SRC emacs-lisp
  (use-package python-mode
    :ensure t
    :commands (python-mode)
    :mode (("\\.py\\'" . python-mode))
    :init
    (setq python-fill-docstring-style 'django)
    (setq fill-column 78)
    :config
    (setq python-shell-interpreter "ipython3")
    (setq python-fill-docstring-style 'django)
    (setq fill-column 78)
    ;; stop python interpreter from clobbering find-file...
    ;; https://answers.launchpad.net/python-mode/+question/221144
    (add-hook 'python-mode-hook
              (lambda ()
                ;; remove python-mode's ffap things that slow down find-file
                (setq ffap-alist
                      (remove '(python-mode . py-ffap-module-path)
                              ffap-alist))
                (setq ffap-alist
                      (remove '(python-mode . py-module-path)
                              ffap-alist))
                (setq ffap-alist
                      (remove '(inferior-python-mode . py-ffap-module-path)
                              ffap-alist)))))
#+END_SRC
