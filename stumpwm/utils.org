* ENCOURAGEMENT

/Scientifically-proven optimal words of hackerish encouragement./

Inspired my the messages printed when starting Emacs' slime mode.

Strings lifted from [[https://github.com/kovisoft/slimv/blob/master/slime/slime.el][here]] and [[https://github.com/kovisoft/slimv/blob/master/slime/slime.el][here]].

#+BEGIN_SRC common-lisp
  (defvar words-of-encouragement
    '("Let the hacking commence!"
      "Hacks and glory await!"
      "Hack and be merry!"
      "Your hacking starts... NOW!"
      "May the source be with you!"
      "Take this REPL, brother, and may it serve you well."
      "Lemonodor-fame is but a hack away!"
      (format t "~a, this could be the start of a beautiful program."
       (string-capitalize (getenv "USER")))
      "Scientifically-proven optimal words of hackerish encouragement."))

  (defun my/random-elt (list)
    "Return a random element from a list."
    (elt list (random (length list))))

  (defun random-encouragement ()
    "Return a random string of hackerish encouragement."
    (my/random-elt words-of-encouragement))

  (defcommand stump-encouragement () ()
    "Print a random string of hackerish encouragement."
    (message "~a" (random-encouragement)))
#+END_SRC

    "creates a swank server in the stumpwm lisp process"
    (setf stumpwm:*top-level-error-action* :break)

* PROMPT

Taken from the example rc file.

Prompt the user for an interactive command. The first arg is an
optional initial contents.

#+BEGIN_SRC common-lisp
  (defcommand prompt (&optional (initial "")) (:rest)
    "prompt the user for an interactive command. The first arg is an optional
     initial contents."
    (let ((cmd (read-one-line (current-screen) ": " :initial-input initial)))
      (when cmd
        (eval-command cmd t))))
#+END_SRC

* VOLUME

Could also pipe to awk and uniq to achieve same goal, but it was more fun doing
it in lisp :-)

#+BEGIN_SRC common-lisp :tangle no
  (defun get-volume (action)
    (let ((output
           (run-shell-command
            (concat "amixer set Master " action "| awk '/%/ {print $5 $6}' | uniq")
            t)))
      (split-string output "][")))
#+END_SRC

#+BEGIN_SRC common-lisp
  (defun get-volume (device action)
    "Return a cleaned up version of the output of the amixer command with a given
       ACTION. Simply return the volume percentage and the status of the device."
    (let* ((output (run-shell-command (concat "amixer set " device " " action) t))
           (start (search "[" output))
           (end (search (concat "]" (string #\linefeed)) output))
           (out (subseq output start end)))
      (split-string out "] [")))

  (defcommand volume (device action)
      ((:string "Enter Device: ")
       (:string "Enter Action: "))
    "Wrap volume commands and print a message containing current volume."
    (let* ((output (get-volume device action))
           (volume (nth 0 output))
           (status (string-upcase (nth 1 output))))
      (message "^B^5VOLUME: ^n~a~%^5STATUS: ^n~a" volume status)))
#+END_SRC

* SENSORS

#+BEGIN_SRC bash :tangle no
  sensors | awk -F '[(:]' '{gsub(/\s+/," ")} {gsub(/+/,"")} /Core|Exhaust/ {printf $2} END {print ""}'
#+END_SRC

#+BEGIN_SRC common-lisp
  (defun get-cpu (output)
    (let* ((start (search "Package id 0:" output))
           (end (search "(high" output))
           (cpu (subseq output start end))
           (cpu (string-trim "Package id 0: +" cpu)))
      cpu))

  (defun get-fan (output)
    (let* ((start (search "Exhaust" output))
           (end (search "(min" output))
           (fan (subseq output start end))
           (fan (string-trim "Exhaust :" fan)))
      fan))

  (defun get-sensors (&optional return-as-string)
    (let* ((output (run-shell-command "sensors" t))
           (cpu (get-cpu output))
           (fan (get-fan output)))
      (if return-as-string
          (concat cpu " " fan)
          (list cpu fan))))

  (defcommand sensors () ()
    (let* ((my-sensors (get-sensors))
           (cpu (nth 0 my-sensors))
           (fan (nth 1 my-sensors)))
      (message "^B^5CPU: ^n~a~%^B^5FAN: ^n~a" cpu fan)))
#+END_SRC
