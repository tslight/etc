* EMACS MAP

#+BEGIN_SRC common-lisp
(defvar *my-emacs-map*
  (let ((m (make-sparse-keymap)))
    (define-key m (kbd "e") "emacs")
    (define-key m (kbd "C-e") "emacs")
    (define-key m (kbd "n") "exec emacs")
    (define-key m (kbd "c") "exec emacsclient -c -a ''")
    (define-key m (kbd "q") "exec emacs --no-init-file --quick")
    (define-key m (kbd "d") "exec emacs --debug-init")
    (define-key m (kbd "t") "exec emacs -nw")
    m ; NOTE: this is important
    ))
#+END_SRC

* TERMINAL MAP

#+BEGIN_SRC common-lisp
(defvar *my-term-map*
  (let ((m (make-sparse-keymap)))
    (define-key m (kbd "C-t") "exec x-terminal-emulator -e tmux attach")
    (define-key m (kbd "s") "prompt exec x-terminal-emulator -e ssh ")
    (define-key m (kbd "r") "prompt exec x-terminal-emulator -e ")
    m ; NOTE: this is important
    ))
#+END_SRC
