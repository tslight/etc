Stolen from [[https://github.com/alezost/stumpwm-config/blob/master/init.lisp][here]]

#+BEGIN_SRC common-lisp
  (defvar my/init-directory
    (directory-namestring
     (truename (merge-pathnames (user-homedir-pathname) ".stumpwm.d")))
    "Path to configuration directory.")

  (defun my/load (filename)
    "Load a file FILENAME (without extension) from `my/init-directory'."
    (let ((file (merge-pathnames (concat filename ".lisp")
                                 my/init-directory)))
      (if (probe-file file)
          (load file)
          (format *error-output* "File '~a' doesn't exist." file))))

  (redirect-all-output (merge-pathnames "log" my/init-directory))

  (my/load "settings")
  (my/load "utils")
  (my/load "keys")

#+END_SRC
