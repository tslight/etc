~~defaults read -g NSUserKeyEquivalents~~

      Copy = "~c";
      Cut = "~x";
      "New Tab" = "~t";
      "Close Tab" = "~w";
      "Open File..." = "~o";
      "Open Location" = "~l";
      "Open Location..." = "~l";
      Paste = "~v";
      "Re-open Closed Tab" = "~$t";
      Redo = "~y";
