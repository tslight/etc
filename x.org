scaling going on...

#+begin_src conf :tangle (if (> (string-to-number (shell-command-to-string "xdpyinfo | grep dimensions | awk '{print $2}' | awk -Fx '{print $1}'")) 2000) "~/.Xdefaults" "no")

  Xcursor*size: 30
  Xcursor*theme: whiteglass

* XSESSION

#+begin_src conf :tangle (if (eq 0 (shell-command "grep -iq lenovo /var/log/messages")) "~/.xsession" "no")
  xinput set-prop "/dev/wsmouse" "WS Pointer Wheel Emulation" 1
  xinput set-prop "/dev/wsmouse" "WS Pointer Wheel Emulation Button" 2
  xinput set-prop "/dev/wsmouse" "WS Pointer Wheel Emulation Axes" 6 7 4 5
  setxkbmap -option 'caps:swapescape,altwin:ctrl_alt_win'
  xmodmap -e "keysym Print = Alt_L"
#+end_src
